**C) JavaScript -- define function and wire it:**

1. Define `initHamburgerToggle` as a standalone function in the `<script>` block (place it right after the `renderSideNav()` function definition, around line 2560):

```javascript
    function initHamburgerToggle() {
      const hamburger = document.getElementById('hamburgerBtn');
      const sideNav = document.getElementById('sideNav');
      const overlay = document.getElementById('sidebarOverlay');
      if (!hamburger || !sideNav) return;

      function openSidebar() {
        sideNav.classList.add('side-nav-open');
        if (overlay) overlay.classList.add('active');
        hamburger.setAttribute('aria-label', 'Navigation schliessen');
        hamburger.innerHTML = '&#10005;';
      }

      function closeSidebar() {
        sideNav.classList.remove('side-nav-open');
        if (overlay) overlay.classList.remove('active');
        hamburger.setAttribute('aria-label', 'Navigation oeffnen');
        hamburger.innerHTML = '&#9776;';
      }

      hamburger.addEventListener('click', function() {
        if (sideNav.classList.contains('side-nav-open')) {
          closeSidebar();
        } else {
          openSidebar();
        }
      });

      if (overlay) {
        overlay.addEventListener('click', closeSidebar);
      }

      sideNav.addEventListener('click', function(e) {
        if (e.target.closest('.side-nav-item')) {
          if (window.innerWidth < 768) {
            closeSidebar();
          }
        }
      });

      document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape' && sideNav.classList.contains('side-nav-open')) {
          closeSidebar();
        }
      });
    }
```
